const axios = require('axios'); // 카카오 모빌리티 API 키 const apiKey = 'YOUR_KAKAO_MOBILITY_API_KEY'; // 출발지, 도착지, 경유지 설정 const start = '출발지'; const destination = '도착지'; const waypoints = ['경유지1', '경유지2']; // 필요한 경우 더 많은 경유지를 추가할 수 있습니다. // 카카오 모빌리티 길찾기 API 엔드포인트 const apiUrl = 'https://api.kakaomobility.com/v1/directions'; // 요청 데이터 구성 const requestData = { sx: start, sy: '출발지 위도', ex: destination, ey: '도착지 위도', waypoints: waypoints.map((waypoint, index) => ({ px: waypoint, py: '경유지 위도', })), }; // 카카오 모빌리티 API를 호출하여 길찾기 정보를 가져옵니다. axios .get(apiUrl, { params: requestData, headers: { Authorization: `KakaoAK ${apiKey}`, }, }) .then((response) => { const data = response.data; // 결과 데이터를 처리합니다. console.log(data); }) .catch((error) => { console.error('API 호출 중 오류 발생:', error); });
